This is an exmaple that shows the basic usage of the AlsaCollectorNode node, the SelectorNode node, the HybridNode node and the ReSpeaker supervisor.

#include <cstring>
#include <memory>
#include <iostream>
#include <csignal>
#include <chrono>
#include <thread>
#include <respeaker.h>
#include <chain_nodes/alsa_collector_node.h>
#include <chain_nodes/selector_node.h>
#include <chain_nodes/hybrid_node.h>
extern "C"
{
#include <sndfile.h>
#include <unistd.h>
#include <getopt.h>
}
using namespace std;
using namespace respeaker;
static bool stop = false;
void SignalHandler(int signal){
cerr << "Caught signal " << signal << ", terminating..." << endl;
stop = true;
//maintain the main thread untill the worker thread released its resource
//std::this_thread::sleep_for(std::chrono::seconds(1));
}
static void help(const char *argv0) {
cout << "alsa_agc_test [options]" << endl;
cout << "A demo application for librespeaker." << endl << endl;
cout << " -h, --help Show this help" << endl;
cout << " -s, --source=SOURCE_NAME The source (microphone) to connect to" << endl;
cout << " -g, --gain=GAIN_DB The agc gain, [-31,0]" << endl;
cout << " -t, --type=AGC_TYPE The agc type, [1,3], 1 for Digital_Fixed, 2 for Digital_Adaptive, 3 for Analog" << endl;
}
int main(int argc, char *argv[]) {
// Configures signal handling.
struct sigaction sig_int_handler;
sig_int_handler.sa_handler = SignalHandler;
sigemptyset(&sig_int_handler.sa_mask);
sig_int_handler.sa_flags = 0;
sigaction(SIGINT, &sig_int_handler, NULL);
sigaction(SIGTERM, &sig_int_handler, NULL);
// parse opts
int c;
string source = "default";
int gain = -10;
int agc_type = 3;
static const struct option long_options[] = {
{"help", 0, NULL, 'h'},
{"source", 1, NULL, 's'},
{"gain", 1, NULL, 'g'},
{"type", 1, NULL, 't'},
{NULL, 0, NULL, 0}
};
while ((c = getopt_long(argc, argv, "s:g:t:h", long_options, NULL)) != -1) {
switch (c) {
case 'h' :
help(argv[0]);
return 0;
case 's':
source = string(optarg);
break;
case 'g':
gain = stoi(optarg);
break;
case 't':
agc_type = stoi(optarg);
break;
default:
return 0;
}
}
unique_ptr<AlsaCollectorNode> collector;
unique_ptr<SelectorNode> selector;
unique_ptr<HybridNode> hybrid;
unique_ptr<ReSpeaker> respeaker;
// rate: 48k len_ms:8 output_interleaved: false
collector.reset(AlsaCollectorNode::Create(source, 48000, 8, false));
std::vector<int> selected_channels {0, 1, 2, 3, 4, 5, };
// selector node output deinterleaved data by default
selector.reset(SelectorNode::Create(selected_channels));
if (agc_type < 1 || agc_type > 3) {
cout << "Invalid agc type: " << agc_type << ", should be [1,3]" << endl;
return -1;
}
// true: output interleaved
hybrid.reset(HybridNode::CreateAgcOnly(agc_type, (int)std::abs(gain), true));
selector->Uplink(collector.get());
hybrid->Uplink(selector.get());
respeaker.reset(ReSpeaker::Create());
respeaker->RegisterChainByHead(collector.get());
respeaker->RegisterOutputNode(hybrid.get());
if (!respeaker->Start(&stop)) {
cout << "Can not start the respeaker node chain." << endl;
return -1;
}
string data;
int frames;
uint16_t tick = 0;
size_t num_channels = respeaker->GetNumOutputChannels();
int rate = respeaker->GetNumOutputRate();
cout << "num channels: " << num_channels << ", rate: " << rate << endl;
// init libsndfile
SNDFILE *file ;
SF_INFO sfinfo ;
memset (&sfinfo, 0, sizeof (sfinfo)) ;
sfinfo.samplerate = rate ;
sfinfo.channels = num_channels ;
sfinfo.format = (SF_FORMAT_WAV | SF_FORMAT_PCM_16) ;
if (! (file = sf_open ("alsa_agc_test.wav", SFM_WRITE, &sfinfo)))
{
cout << "Error : Not able to open output file." << endl;
return -1 ;
}
while (!stop)
{
data = respeaker->Listen();
frames = data.length() / (sizeof(int16_t) * num_channels);
sf_writef_short(file, (const int16_t *)(data.data()), frames);
if (tick++ % 100 == 0) {
std::cout << "collector: " << collector->GetQueueDeepth() << ", selector: " <<
selector->GetQueueDeepth() << ", hybrid: " << hybrid->GetQueueDeepth() <<
std::endl;
}
}
cout << "stopping the respeaker worker thread..." << endl;
respeaker->Stop();
cout << "cleanup done." << endl;
sf_close (file);
return 0;
}
